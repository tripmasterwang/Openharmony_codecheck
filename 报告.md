def call_model(base_url, api_key, model, messages):
    url = f"{base_url}/v1/chat/completions"
    headers = {
        "Authorization": f"Bearer {api_key}",
        "Content-Type": "application/json"
    }
    data = {
        "model": model,
        "messages": messages
    }
    resp = requests.post(url, json=data, headers=headers)
    return resp.json()
